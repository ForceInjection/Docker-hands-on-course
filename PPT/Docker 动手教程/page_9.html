Dockerfile 基础

FROM
指定基础镜像，所有构建的第一条指令
FROM nginx:alpine
LABEL
添加镜像元数据，如维护者信息
LABEL maintainer="your-email@example.com"
WORKDIR
设置工作目录，影响后续指令
WORKDIR /usr/share/nginx/html
COPY
从构建上下文复制文件到镜像
COPY index.html .
ADD
复制文件，支持URL和自动解压
ADD app.tar.gz /app/
RUN
执行命令并创建新的镜像层
RUN apk add --no-cache curl
ENV
设置环境变量
ENV NODE_ENV=production
EXPOSE
声明容器监听的端口
EXPOSE 80 443
VOLUME
创建挂载点，用于数据持久化
VOLUME ["/data"]
CMD
容器启动时执行的默认命令
CMD ["nginx", "-g", "daemon off;"]
最佳实践： 使用多阶段构建减小镜像大小，合理安排指令顺序以优化缓存，每个RUN执行多个相关命令减少层数。
